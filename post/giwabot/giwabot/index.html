giwabot@raspberrypi:/usr/lib/systemd/system $ cd /usr/lib/systemd/system/
giwabot@raspberrypi:/usr/lib/systemd/system $ sudo vi giwabot.service
 [Unit]
 Description=Giwabot

 [Service]
 Type=forking
 User=giwabot
 Group=giwabot
 ExecStart=/home/giwabot/Documents/solar-cleanbot-vision/onbootstartup.sh

 [Install]
 WantedBy=multi-user.target

 giwabot@raspberrypi:/usr/lib/systemd/system $ sudo systemctl enable giwabot  #开机自动启动
  2. 后续常用命令
    1. 开机自动运行后，关闭服务 sudo systemctl stop giwabot
    2. 设置开机不自动启动 sudo systemctl disable giwabot
    3. 手动开启服务 sudo systemctl start/restart giwabot
